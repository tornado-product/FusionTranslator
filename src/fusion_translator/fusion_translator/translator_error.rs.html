3/// 翻译模块错误类型
4///
5/// 枚举定义了在翻译过程中可能遇到的各种错误情况
6#[derive(Debug, thiserror::Error)]
7#[allow(dead_code)]
8pub enum TranslatorError {
9    /// 网络请求失败
10    ///
11    /// 发送HTTP请求时发生错误，可能是网络连接问题或服务器不可用
12    #[error("Failed to fetch")]
13    Reqwest(#[from] reqwest::Error),
14    /// API返回错误响应
15    ///
16    /// 翻译API返回了错误响应，可能是认证失败、参数错误等
17    #[error("Api returned invalid response")]
18    ApiError(ApiError),
19    /// 未知的目标语言
20    ///
21    /// 指定的目标语言无法被识别或不支持
22    #[error("Couldnt convert language")]
23    UnknownLanguage(Language),
24    /// 语言代码映射失败
25    ///
26    /// 无法将字符串转换为有效的Language枚举值
27    #[error("failed to convert string to language")]
28    CouldNotMapLanguage(Option<String>),
29    /// 未收到API响应
30    ///
31    /// 发送请求后未收到任何响应，可能是网络超时或服务器无响应
32    #[error("api did not return a response")]
33    NoResponse,
34    /// 请求文本过长
35    ///
36    /// 发送的翻译请求超过了API允许的最大长度限制
37    ///
38    /// # 参数
39    /// - 第一个u32: 实际请求长度
40    /// - 第二个u32: 最大允许长度
41    #[error("Request was too long")]
42    RequestToLong(u32, u32),
43    /// 请求失败
44    ///
45    /// HTTP请求返回了错误的响应状态码
46    ///
47    /// # 参数
48    /// - u16: HTTP状态码
49    #[error("Request failed with status code")]
50    RequestFailed(u16),
51    /// 缺少源语言参数
52    ///
53    /// 某些翻译API需要明确指定源语言，但调用时未提供
54    #[error("Translator required a input language")]
55    NoLanguage,
56}
57
58/// API错误详细信息
59///
60/// 包含特定翻译API返回的详细错误信息
61#[derive(Debug)]
62#[allow(dead_code)]
63pub enum ApiError {
64    /// 百度翻译API错误
65    ///
66    /// 包含错误代码和错误消息
67    ///
68    /// # 字段
69    /// - `code`: 百度API返回的错误代码
70    /// - `message`: 错误描述信息
71    Baidu { code: String, message: String },
72}
73
74impl std::fmt::Display for ApiError {
75    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
76        match self {
77            ApiError::Baidu { code, message } => {
78                write!(f, "Baidu API Error [{}]: {}", code, message)
79            }
80        }
81    }
