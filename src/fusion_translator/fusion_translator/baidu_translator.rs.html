6/// 百度翻译器实现
7///
8/// 通过调用百度翻译API实现文本翻译功能
9pub struct BaiduTranslator {
10    /// API请求地址
11    url: String,
12    /// 百度开放平台应用ID
13    app_id: String,
14    /// 百度开放平台应用密钥
15    key: String,
16    /// HTTP客户端
17    client: Client,
18}
19
20#[async_trait::async_trait]
21impl AsyncTranslator for BaiduTranslator {
22    /// 判断是否为本地翻译器
23    ///
24    /// 百度翻译器需要调用远程API，返回false
25    fn local(&self) -> bool {
26        false
27    }
28
29    /// 翻译单个文本
30    ///
31    /// # 参数
32    /// - `query`: 待翻译的文本
33    /// - `from`: 源语言，None表示自动检测
34    /// - `to`: 目标语言
35    ///
36    /// # 返回值
37    /// 翻译结果
38    async fn translate(
39        &self,
40        query: &str,
41        from: Option<Language>,
42        to: &Language,
43    ) -> anyhow::Result<TranslationOutput> {
44        let to = to.to_baidu().ok_or(TranslatorError::UnknownLanguage(*to))?;
45        let from = match from {
46            Some(item) => item.to_baidu().ok_or(TranslatorError::UnknownLanguage(item))?,
47            None => "auto",
48        };
49        let form = Form::new(&self.app_id, query, "0", &self.key, from, to);
50        let resp: Response = self
51            .client
52            .post(&self.url)
53            .form(&form)
54            .send()
55            .await?
56            .json()
57            .await?;
58        let resp = match resp {
59            Response::Ok(v) => v,
60            Response::Err(v) => {
61                Err(TranslatorError::ApiError(ApiError::Baidu {
62                    message: v.solution().to_owned(),
63                    code: v.code,
64                }))?;
65                unreachable!()
66            }
67        };
68        Ok(TranslationOutput {
69            text: resp
70                .trans_result
71                .iter()
72                .map(|v| v.dst.to_string())
73                .collect::<Vec<_>>()
74                .join("\n"),
75            lang: Some(
76                Language::from_baidu(&resp.to).ok_or(TranslatorError::CouldNotMapLanguage(Some(resp.to)))?,
77            ),
78        })
79    }
80
81    /// 翻译多个文本
82    ///
83    /// # 参数
84    /// - `query`: 待翻译的文本数组
85    /// - `from`: 源语言，None表示自动检测
86    /// - `to`: 目标语言
87    ///
88    /// # 返回值
89    /// 翻译结果列表
90    async fn translate_vec(
91        &self,
92        query: &[String],
93        from: Option<Language>,
94        to: &Language,
95    ) -> anyhow::Result<TranslationListOutput> {
96        let v = self.translate(&query.join("\n"), from, to).await?;
97        Ok(TranslationListOutput {
98            text: v.text.split('\n').map(|v| v.to_string()).collect(),
99            lang: v.lang,
100        })
101    }
102}

124/// 表单数据提交结构
125///
126/// 用于构造百度翻译API的请求参数
127#[derive(Debug, Serialize)]
128pub struct Form {
129    /// 待翻译文本
130    pub q: String,
131    /// 源语言
132    pub from: String,
133    /// 目标语言
134    pub to: String,
135    /// 应用ID
136    pub appid: String,
137    /// 随机盐值
138    pub salt: String,
139    /// 签名
140    pub sign: String,
141}
142
143impl Form {
144    /// 创建新的表单实例
145    ///
146    /// # 参数
147    /// - `appid`: 百度开放平台应用ID
148    /// - `q`: 待翻译文本
149    /// - `salt`: 随机盐值
150    /// - `key`: 百度开放平台应用密钥
151    /// - `from`: 源语言代码
152    /// - `to`: 目标语言代码
153    ///
154    /// # 返回值
155    /// 新的表单实例
156    #[allow(dead_code)]
157    fn new(appid: &str, q: &str, salt: &str, key: &str, from: &str, to: &str) -> Self {
158        let data = format!("{}{}{}{}", &appid, q, salt, &key);
159        let sign = format!("{:x}", md5::compute(data));
160        Self {
161            q: q.to_string(),
162            from: from.to_string(),
163            to: to.to_string(),
164            appid: appid.to_string(),
165            salt: salt.to_string(),
166            sign,
167        }
168    }
169}

184/// 百度API错误信息
185///
186/// 包含错误代码和错误消息
187#[derive(Debug, Clone, Deserialize)]
188pub struct BaiduApiError {
189    /// 错误代码
190    #[serde(rename = "error_code")]
191    pub code: String,
192    /// 错误消息
193    #[serde(rename = "error_msg")]
194    pub msg: String,
195}
196
197impl std::error::Error for BaiduApiError {}
198impl std::fmt::Display for BaiduApiError {
199    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
200        write!(
201            f,
202            "Error code: `{}`\nError message: `{}`\nError meaning: {}\nThe above content is returned by Baidu translation API",
203            self.code,
204            self.msg,
205            self.solution()
206        )
207    }
208}
